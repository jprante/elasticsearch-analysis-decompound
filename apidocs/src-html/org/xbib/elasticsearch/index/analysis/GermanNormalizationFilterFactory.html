020    package org.xbib.elasticsearch.index.analysis;
021    
022    import org.apache.lucene.analysis.TokenStream;
023    import org.apache.lucene.analysis.de.GermanNormalizationFilter;
024    import org.elasticsearch.common.inject.Inject;
025    import org.elasticsearch.common.inject.assistedinject.Assisted;
026    import org.elasticsearch.common.settings.Settings;
027    import org.elasticsearch.index.Index;
028    import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
029    import org.elasticsearch.index.settings.IndexSettings;
030    
031    /**
032     *
033     */
034    public class GermanNormalizationFilterFactory extends AbstractTokenFilterFactory {
035    
036        @Inject
037        public GermanNormalizationFilterFactory(Index index, @IndexSettings Settings indexSettings, @Assisted String name, @Assisted Settings settings) {
038            super(index, indexSettings, name, settings);
039        }
040    
041        @Override
042        public TokenStream create(TokenStream tokenStream) {
043            return new GermanNormalizationFilter(tokenStream);
044        }
045    }
