016package org.xbib.elasticsearch.analysis.decompound;
017
018import java.io.IOException;
019import java.io.InputStream;
020import java.util.ArrayList;
021import java.util.List;
022import java.util.StringTokenizer;
023
024public class Decompounder {
025
026    private CompactPatriciaTrie kompvvTree;
027    private CompactPatriciaTrie kompvhTree;
028    private CompactPatriciaTrie grfTree;
029
030    public Decompounder(InputStream kompvv, InputStream kompvh, InputStream gfred, double threshold)
031            throws IOException, ClassNotFoundException {
032        kompvvTree = new CompactPatriciaTrie();
033        kompvvTree.load(kompvv);
034        kompvvTree.setIgnoreCase(true);
035        kompvvTree.setThreshold(threshold);
036        kompvhTree = new CompactPatriciaTrie();
037        kompvhTree.load(kompvh);
038        kompvhTree.setIgnoreCase(true);
039        kompvhTree.setThreshold(threshold);
040        grfTree = new CompactPatriciaTrie();
041        grfTree.load(gfred);
042        grfTree.setIgnoreCase(true);
043        grfTree.setThreshold(threshold); // previous value = 0.46
044    }
045
046    public Decompounder(CompactPatriciaTrie kompvv, CompactPatriciaTrie kompvh, CompactPatriciaTrie gfred) {
047        kompvvTree = kompvv;
048        kompvhTree = kompvh;
049        grfTree = gfred;
050    }
051    
052    private String reverse(String torev) {
053        String ret = "";
054        for (int i = torev.length(); i > 0; i--) {
055            ret += torev.substring(i - 1, i);
056        }
057        return ret;
058    }
059
060    public List<String> decompound(String word) {
061        word = reduceToBaseForm(word);
062        List<String> list = new ArrayList<String>();
063        String classvv = kompvvTree.classify(word + "<");
064        String classvh = kompvhTree.classify(reverse(word) + "<");
065        StringBuilder numStrvv = new StringBuilder();
066        StringBuilder numStrvh = new StringBuilder();
067        StringBuilder suffixvv = new StringBuilder();
068        StringBuilder suffixvh = new StringBuilder();
069        String vvpart1 = "";
070        String vhpart1 = "";
071        String vvpart2 = "";
072        String vhpart2 = "";
073        int numvv = 0;
074        int numvh = 0;
075        boolean vhOk = true;
076        boolean vvOk = true;
077        if (classvv.equals("undecided")) {
078            vvOk = false;
079        }
080        if (classvh.equals("undecided")) {
081            vhOk = false;
082        }
083        if (vvOk) {
084            for (int i = 0; i < classvv.length(); i++) {
085                char c = classvv.charAt(i);
086                if ((c <= '9') && (c >= '0')) {
087                    numStrvv.append(c);
088                } else {
089                    suffixvv.append(c);
090                }
091            }
092        }
093        if (vhOk) {
094            for (int i = 0; i < classvh.length(); i++) {
095                char c = classvh.charAt(i);
096                if ((c <= '9') && (c >= '0')) {
097                    numStrvh.append(c);
098                } else {
099                    suffixvh.append(c);
100                }
101            }
102        }
103
104        if (vvOk) {
105            numvv = Integer.parseInt(numStrvv.toString());
106        }
107        if (vhOk) {
108            numvh = Integer.parseInt(numStrvh.toString());
109        }
110
111        if (vvOk) {
112            if (numvv >= word.length()) {
113                vvOk = false;
114            }
115        }
116
117        if (vhOk) {
118            if (numvh >= word.length()) {
119                vhOk = false;
120            }
121        }
122
123        if (vvOk) {
124            for (int i = 0; i < suffixvv.length(); i++) {
125                if (word.length() > (numvv + i)) {
126                    if (suffixvv.charAt(i) != word.charAt(numvv + i)) {
127                        vvOk = false;
128                    }
129                } else {
130                    vvOk = false;
131                }
132            }
133        }
134        if (vhOk) {
135            for (int i = 0; i < suffixvh.length(); i++) {
136                if (suffixvh.charAt(i) != word.charAt(numvh + 1 + i)) {
137                    vvOk = false;
138                }
139            }
140        }
141        if (vvOk) {
142            vvpart1 = word.substring(0, numvv);
143            vvpart2 = word.substring(numvv + suffixvv.length(), word.length());
144            if (vvpart2.length() <= 3) {
145                vvOk = false;
146            }
147        }
148        if (vhOk) {
149            vhpart1 = word.substring(0, word.length() - numvh);
150            vhpart2 = word.substring(word.length() - (numvh + suffixvh.length()), word.length());
151            if (vhpart1.length() <= 3) {
152                vhOk = false;
153            }
154
155        }
156        if (vvOk && vhOk) {
157            if (vvpart1.equals(vhpart1)) {
158                list.add(vvpart1);
159                if (vhpart2.length() < vvpart2.length()) {
160                    list.add(vhpart2);
161                } else if (vhpart2.length() > vvpart2.length()) {
162                    list.add(vvpart2);
163                }
164            } else if ((vhpart1.length() - vvpart1.length()) < 3) {
165                list.add(vvpart1);
166                if (vhpart2.length() < vvpart2.length()) {
167                    list.add(vhpart2);
168                } else if (vhpart2.length() > vvpart2.length()) {
169                    list.add(vvpart2);
170                }
171            }
172            else {
173                list.add(vvpart1);
174                list.add(word.substring(vvpart1.length() + suffixvv.length(), word.length() - numvh));
175                list.add(vhpart2);
176            }
177            if (vvpart2.equals(vhpart2)) {
178                list.add(vvpart2);
179            }
180
181        } else if (vvOk && !vhOk) {
182            list.add(vvpart1);
183            list.add(vvpart2);
184        } else if (vhOk && !vvOk) {
185            list.add(vhpart1);
186            list.add(vhpart2);
187        } else {
188            list.add(word);
189        }
190        List<String> retvec2 = new ArrayList();
191        List<String> l;
192        if (list.size() > 1) {
193            for (String s : list) {
194                l = decompound(s);
195                retvec2.addAll(l);
196            }
197        }
198        else {
199            retvec2 = list;
200        }
201        return retvec2;
202    }
203
204    public String reduceToBaseForm(String word) {
205        String result = word;
206        String baseForm = grfTree.classify(reverse(word));
207        if (!baseForm.equals("undecided")) {
208            StringTokenizer st = new StringTokenizer(baseForm, ",");
209            baseForm = st.nextToken();
210            StringBuilder numStr = new StringBuilder();
211            StringBuilder suffix = new StringBuilder();
212            for (int i = 0; i < baseForm.length(); i++) {
213                char c = baseForm.charAt(i);
214                if ((c <= '9') && (c >= '0')) {
215                    numStr.append(c);
216                } else {
217                    suffix.append(c);
218                }
219            }
220            if (numStr.length() > 0) {
221                int cutpos = Integer.parseInt(numStr.toString());
222                if (cutpos > result.length()) {
223                    cutpos = result.length();
224                }
225                result = result.substring(0, result.length() - cutpos) + suffix;
226            }
227        }
228        return result;
229    }
