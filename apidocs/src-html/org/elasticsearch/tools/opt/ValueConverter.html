026package org.elasticsearch.tools.opt;
027
028/**
029 * Instances of this interface are used to convert arguments of options into specific Java types.
030 *
031 * @param <V> constraint on the type of values being converted to
032 * @author <a href="mailto:pholser@alumni.rice.edu">Paul Holser</a>
033 */
034public interface ValueConverter<V> {
035    /**
036     * Converts the given string value into a Java type.
037     *
038     * @param value the string to convert
039     * @return the converted value
040     * @throws ValueConversionException if a problem occurs while converting the value
041     */
042    V convert( String value );
043
044    /**
045     * Gives the class of the type of values this converter converts to.
046     *
047     * @return the target class for conversion
048     */
049    Class<V> valueType();
050
051    /**
052     * Gives a string that describes the pattern of the values this converter expects, if any.  For example, a date
053     * converter can respond with a {@link java.text.SimpleDateFormat date format string}.
054     *
055     * @return a value pattern, or {@code null} if there's nothing interesting here
056     */
057    String valuePattern();
058}
