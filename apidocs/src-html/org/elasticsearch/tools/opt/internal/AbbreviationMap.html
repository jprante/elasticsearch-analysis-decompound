026package org.elasticsearch.tools.opt.internal;
027
028import java.util.Map;
029import java.util.TreeMap;
030
031/**
032 * <p>A map whose keys are strings; when a key/value pair is added to the map,
033 * the longest unique abbreviations of that key are added as well, and associated with
034 * the value.  Thus:</p>
035 *
036 * <pre>
037 *   <code>
038 *   abbreviations.put( "good", "bye" );
039 *   </code>
040 * </pre>
041 *
042 * <p>would make it such that you could retrieve the value {@code "bye"} from the map
043 * using the keys {@code "good"}, {@code "goo"}, {@code "go"}, and {@code "g"}.
044 * A subsequent invocation of:</p>
045 * <pre>
046 *   <code>
047 *   abbreviations.put( "go", "fish" );
048 *   </code>
049 * </pre>
050 *
051 * <p>would make it such that you could retrieve the value {@code "bye"} using the keys
052 * {@code "good"} and {@code "goo"}, and the value {@code "fish"} using the key
053 * {@code "go"}.  The key {@code "g"} would yield {@code null}, since it would no longer
054 * be a unique abbreviation.</p>
055 *
056 * <p>The data structure is much like a "trie".</p>
057 *
058 * @param <V> a constraint on the types of the values in the map
059 * @author <a href="mailto:pholser@alumni.rice.edu">Paul Holser</a>
060 * @see <a href="http://www.perldoc.com/perl5.8.0/lib/Text/Abbrev.html">Perl's Text::Abbrev module</a>
061 */
062public class AbbreviationMap<V> {
063    private String key;
064    private V value;
065    private final Map<Character, AbbreviationMap<V>> children = new TreeMap<Character, AbbreviationMap<V>>();
066    private int keysBeyond;
067
068    /**
069     * <p>Tells whether the given key is in the map, or whether the given key is a unique
070     * abbreviation of a key that is in the map.</p>
071     *
072     * @param aKey key to look up
073     * @return {@code true} if {@code key} is present in the map
074     * @throws NullPointerException if {@code key} is {@code null}
075     */
076    public boolean contains( String aKey ) {
077        return get( aKey ) != null;
078    }
079
080    /**
081     * <p>Answers the value associated with the given key.  The key can be a unique
082     * abbreviation of a key that is in the map. </p>
083     *
084     * @param aKey key to look up
085     * @return the value associated with {@code aKey}; or {@code null} if there is no
086     * such value or {@code aKey} is not a unique abbreviation of a key in the map
087     * @throws NullPointerException if {@code aKey} is {@code null}
088     */
089    public V get( String aKey ) {
090        char[] chars = charsOf( aKey );
091
092        AbbreviationMap<V> child = this;
093        for ( char each : chars ) {
094            child = child.children.get( each );
095            if ( child == null )
096                return null;
097        }
098
099        return child.value;
100    }
101
102    /**
103     * <p>Associates a given value with a given key.  If there was a previous
104     * association, the old value is replaced with the new one.</p>
105     *
106     * @param aKey key to create in the map
107     * @param newValue value to associate with the key
108     * @throws NullPointerException if {@code aKey} or {@code newValue} is {@code null}
109     * @throws IllegalArgumentException if {@code aKey} is a zero-length string
110     */
111    public void put( String aKey, V newValue ) {
112        if ( newValue == null )
113            throw new NullPointerException();
114        if ( aKey.length() == 0 )
115            throw new IllegalArgumentException();
116
117        char[] chars = charsOf( aKey );
118        add( chars, newValue, 0, chars.length );
119    }
120
121    /**
122     * <p>Associates a given value with a given set of keys.  If there was a previous
123     * association, the old value is replaced with the new one.</p>
124     *
125     * @param keys keys to create in the map
126     * @param newValue value to associate with the key
127     * @throws NullPointerException if {@code keys} or {@code newValue} is {@code null}
128     * @throws IllegalArgumentException if any of {@code keys} is a zero-length string
129     */
130    public void putAll( Iterable<String> keys, V newValue ) {
131        for ( String each : keys )
132            put( each, newValue );
133    }
134
135    private boolean add( char[] chars, V newValue, int offset, int length ) {
136        if ( offset == length ) {
137            value = newValue;
138            boolean wasAlreadyAKey = key != null;
139            key = new String( chars );
140            return !wasAlreadyAKey;
141        }
142
143        char nextChar = chars[ offset ];
144        AbbreviationMap<V> child = children.get( nextChar );
145        if ( child == null ) {
146            child = new AbbreviationMap<V>();
147            children.put( nextChar, child );
148        }
149
150        boolean newKeyAdded = child.add( chars, newValue, offset + 1, length );
151
152        if ( newKeyAdded )
153            ++keysBeyond;
154
155        if ( key == null )
156            value = keysBeyond > 1 ? null : newValue;
157
158        return newKeyAdded;
159    }
160
161    /**
162     * <p>If the map contains the given key, dissociates the key from its value.</p>
163     *
164     * @param aKey key to remove
165     * @throws NullPointerException if {@code aKey} is {@code null}
166     * @throws IllegalArgumentException if {@code aKey} is a zero-length string
167     */
168    public void remove( String aKey ) {
169        if ( aKey.length() == 0 )
170            throw new IllegalArgumentException();
171
172        char[] keyChars = charsOf( aKey );
173        remove( keyChars, 0, keyChars.length );
174    }
175
176    private boolean remove( char[] aKey, int offset, int length ) {
177        if ( offset == length )
178            return removeAtEndOfKey();
179
180        char nextChar = aKey[ offset ];
181        AbbreviationMap<V> child = children.get( nextChar );
182        if ( child == null || !child.remove( aKey, offset + 1, length ) )
183            return false;
184
185        --keysBeyond;
186        if ( child.keysBeyond == 0 )
187            children.remove( nextChar );
188        if ( keysBeyond == 1 && key == null )
189            setValueToThatOfOnlyChild();
190
191        return true;
192    }
193
194    private void setValueToThatOfOnlyChild() {
195        Map.Entry<Character, AbbreviationMap<V>> entry = children.entrySet().iterator().next();
196        AbbreviationMap<V> onlyChild = entry.getValue();
197        value = onlyChild.value;
198    }
199
200    private boolean removeAtEndOfKey() {
201        if ( key == null )
202            return false;
203
204        key = null;
205        if ( keysBeyond == 1 )
206            setValueToThatOfOnlyChild();
207        else
208            value = null;
209
210        return true;
211    }
212
213    /**
214    * Gives a Java map representation of this abbreviation map.
215     *
216     * @return a Java map corresponding to this abbreviation map
217     */
218    public Map<String, V> toJavaUtilMap() {
219        Map<String, V> mappings = new TreeMap<String, V>();
220        addToMappings( mappings );
221        return mappings;
222    }
223
224    private void addToMappings( Map<String, V> mappings ) {
225        if ( key != null )
226            mappings.put( key, value );
227
228        for ( AbbreviationMap<V> each : children.values() )
229            each.addToMappings( mappings );
230    }
231
232    private static char[] charsOf( String aKey ) {
233        char[] chars = new char[ aKey.length() ];
234        aKey.getChars( 0, aKey.length(), chars, 0 );
235        return chars;
236    }
237}
