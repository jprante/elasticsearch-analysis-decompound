016package org.elasticsearch.plugin.analysis.decompound;
017
018import org.elasticsearch.index.analysis.AnalysisModule;
019import org.elasticsearch.index.analysis.DecompoundTokenFilterFactory;
020import org.elasticsearch.index.analysis.GermanNormalizationFilterFactory;
021import org.elasticsearch.plugins.AbstractPlugin;
022
023public class AnalysisDecompoundPlugin extends AbstractPlugin {
024
025    @Override public String name() {
026        return "analysis-decompound";
027    }
028
029    @Override public String description() {
030        return "A decompounding token filter for german and other languages";
031    }
032
033    public void onModule(AnalysisModule module) {
034        module.addTokenFilter("decompound", DecompoundTokenFilterFactory.class);
035        module.addTokenFilter("german_normalize", GermanNormalizationFilterFactory.class);
036    }
037}
