001package org.xbib.elasticsearch.index.analysis;
002
003import java.io.IOException;
004import java.io.StringReader;
005
006import org.apache.lucene.analysis.TokenStream;
007import org.apache.lucene.analysis.Tokenizer;
008import org.apache.lucene.analysis.standard.StandardTokenizer;
009import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
010import org.apache.lucene.util.Version;
011
012import org.elasticsearch.common.inject.Injector;
013import org.elasticsearch.common.inject.ModulesBuilder;
014import org.elasticsearch.common.settings.ImmutableSettings;
015import org.elasticsearch.common.settings.Settings;
016import org.elasticsearch.common.settings.SettingsModule;
017import org.elasticsearch.env.Environment;
018import org.elasticsearch.env.EnvironmentModule;
019import org.elasticsearch.index.Index;
020import org.elasticsearch.index.IndexNameModule;
021import org.elasticsearch.index.analysis.AnalysisModule;
022import org.elasticsearch.index.analysis.AnalysisService;
023import org.elasticsearch.index.analysis.TokenFilterFactory;
024import org.elasticsearch.index.settings.IndexSettingsModule;
025import org.elasticsearch.indices.analysis.IndicesAnalysisModule;
026import org.elasticsearch.indices.analysis.IndicesAnalysisService;
027
028import org.hamcrest.MatcherAssert;
029import org.hamcrest.Matchers;
030import org.xbib.elasticsearch.plugin.analysis.decompound.AnalysisDecompoundPlugin;
031import org.testng.Assert;
032import org.testng.annotations.Test;
033
034public class GermanNormalizationTests extends Assert {
035
036    @Test
037    public void test() throws IOException {
038        AnalysisService analysisService = createAnalysisService();
039
040        TokenFilterFactory tokenFilter = analysisService.tokenFilter("umlaut");
041        MatcherAssert.assertThat(tokenFilter, Matchers.instanceOf(GermanNormalizationFilterFactory.class));
042
043        String source = "Ein schöner Tag in Köln im Café an der Straßenecke";
044
045        String[] expected = {
046            "Ein",
047            "schoner",
048            "Tag",
049            "in",
050            "Koln",
051            "im",
052            "Café",
053            "an",
054            "der",
055            "Strassenecke"
056        };
057
058        Tokenizer tokenizer = new StandardTokenizer(Version.LUCENE_36, new StringReader(source));
059
060        assertSimpleTSOutput(tokenFilter.create(tokenizer), expected);
061
062    }
063
064    public AnalysisService createAnalysisService() {
065        Settings settings = ImmutableSettings.settingsBuilder().loadFromClasspath("org/xbib/elasticsearch/index/analysis/german_normalization_analysis.json").build();
066
067        Index index = new Index("test");
068
069        Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings),
070                new EnvironmentModule(new Environment(settings)),
071                new IndicesAnalysisModule())
072                .createInjector();
073
074        AnalysisModule analysisModule = new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class));
075        new AnalysisDecompoundPlugin().onModule(analysisModule);
076
077        Injector injector = new ModulesBuilder().add(
078                new IndexSettingsModule(index, settings),
079                new IndexNameModule(index),
080                analysisModule)
081                .createChildInjector(parentInjector);
082
083        return injector.getInstance(AnalysisService.class);
084    }
085
086    public static void assertSimpleTSOutput(TokenStream stream,
087            String[] expected) throws IOException {
088        stream.reset();
089        CharTermAttribute termAttr = stream.getAttribute(CharTermAttribute.class);
090        assertNotNull(termAttr);
091        int i = 0;
092        while (stream.incrementToken()) {
093            assertTrue(i < expected.length);
094            assertEquals(expected[i++], termAttr.toString(), "expected different term at index " + i);
095        }
096        assertEquals(i, expected.length, "not all tokens produced");
097    }
098}
