035public class DecompoundTokenFilterTests {
036
037    @Test
038    public void test() throws IOException {
039        AnalysisService analysisService = createAnalysisService();
040
041
042        TokenFilterFactory tokenFilter = analysisService.tokenFilter("decomp");
043        MatcherAssert.assertThat(tokenFilter, Matchers.instanceOf(DecompoundTokenFilterFactory.class));
044
045        String source = "Die Jahresfeier der Rechtsanwaltskanzleien auf dem Donaudampfschiff hat viel Ökosteuer gekostet";
046
047        String[] expected = {
048            "Die",
049            "Die",
050            "Jahresfeier",
051            "Jahr",
052            "feier",
053            "der",
054            "der",
055            "Rechtsanwaltskanzleien",
056            "Recht",
057            "anwalt",
058            "kanzlei",
059            "auf",
060            "auf",
061            "dem",
062            "dem",
063            "Donaudampfschiff",
064            "Donau",
065            "dampf",
066            "schiff",
067            "hat",
068            "hat",
069            "viel",
070            "viel",
071            "Ökosteuer",
072            "Ökosteuer",
073            "gekostet",
074            "gekosten"
075        };
076
077        Tokenizer tokenizer = new StandardTokenizer(Version.LUCENE_36, new StringReader(source));
078
079        assertSimpleTSOutput(tokenFilter.create(tokenizer), expected);
080
081    }
082
083    public AnalysisService createAnalysisService() {
084        Settings settings = ImmutableSettings.settingsBuilder().loadFromClasspath("org/xbib/elasticsearch/index/analysis/decompound_analysis.json").build();
085
086        Index index = new Index("test");
087
088        Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings),
089                new EnvironmentModule(new Environment(settings)),
090                new IndicesAnalysisModule())
091                .createInjector();
092
093        AnalysisModule analysisModule = new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class));
094        new AnalysisDecompoundPlugin().onModule(analysisModule);
095
096        Injector injector = new ModulesBuilder().add(
097                new IndexSettingsModule(index, settings),
098                new IndexNameModule(index),
099                analysisModule)
100                .createChildInjector(parentInjector);
101
102        return injector.getInstance(AnalysisService.class);
103    }
104
105    public static void assertSimpleTSOutput(TokenStream stream,
106            String[] expected) throws IOException {
107        stream.reset();
108        CharTermAttribute termAttr = stream.getAttribute(CharTermAttribute.class);
109        Assert.assertNotNull(termAttr);
110        int i = 0;
111        while (stream.incrementToken()) {
112            Assert.assertTrue(i < expected.length);
113            Assert.assertEquals(expected[i++], termAttr.toString(), "expected different term at index " + i);
114        }
115        Assert.assertEquals(i, expected.length, "not all tokens produced");
116    }
117}
