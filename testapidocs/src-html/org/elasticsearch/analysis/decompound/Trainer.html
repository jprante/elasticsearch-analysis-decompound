080    public void trainReduce(Reader reader) throws IOException, ClassNotFoundException {
081        reduce = new CompactPatriciaTrie();
082        reduce.setIgnoreCase(true);
083        reduce.setReverse(true);
084        BufferedReader br = new BufferedReader(reader);
085        String line;
086        while ((line = br.readLine()) != null) {
087            if (line.startsWith("#")) {
088                continue;
089            }
090            String[] forms = line.split(" ");
091            if (forms.length == 2) {
092                // <fullform> <baseform>
093                String wclass = createRule(forms[0], forms[1]);
094                //String cl = reduce.classify(line);
095                System.err.println("w=" + forms[0]  + " wclass="+ wclass);
096                reduce.train(forms[0], wclass);
097            }
098        }
099    }

218    public String createRule(String fullform, String baseform) {
219        String rule = "";
220        String part;
221        int bestlength = -1;
222        int start;
223        int end;
224        int occur1;
225        int occur2;
226        int length = fullform.length();
227        for (int i = 0; i < length; i++) {
228            for (int j = length; j > i; j--) {
229                part = fullform.substring(i, j);
230                occur1 = baseform.indexOf(part);
231                occur2 = fullform.indexOf(part);
232                if ((part.length() > bestlength) && occur1 == 0) {
233                    bestlength = part.length();
234                    start = i;
235                    end = j;
236                    rule = "";
237                    if (occur2 != 0) {
238                        rule = fullform.substring(0, start) + "#";
239                    }
240                    rule = rule + (fullform.length() - end) + baseform.substring(part.length(), baseform.length());
241                }
242            }
243        }
244        if (rule.equals("")) {
245            rule = fullform.length() + baseform;
246        }
247        return rule;
248    }

250    public String applyRule(String fullform, String wordClass) {
251        String baseform = "fehlende Behandlung in applyRule";
252        String pattern1 = "[0-9][0-9].*";
253        String pattern2 = "[0-9].*";
254        int temp;
255        if ("undecided".equals(wordClass)) {
256            baseform = "undecided";
257        } else if (wordClass.substring(0, 1).equals("‰")) {
258            temp = Integer.parseInt(wordClass.substring(1, 2));
259            baseform = fullform.substring(0, fullform.length() - temp);
260            baseform = substitute(baseform);
261        } else {
262            int i = wordClass.indexOf("#");
263            if (i != -1) {
264                baseform = fullform.substring(i);
265                if (wordClass.substring(i + 1).matches(pattern1)) {
266                    temp = Integer.parseInt(wordClass.substring(i + 1, i + 3));
267                    baseform = baseform.substring(0, baseform.length() - temp) + wordClass.substring(i + 3);
268                } else if (wordClass.substring(i + 1).matches(pattern2)) {
269                    temp = Integer.parseInt(wordClass.substring(i + 1, i + 2));
270                    baseform = baseform.substring(0, baseform.length() - temp) + wordClass.substring(i + 2);
271                }
272            } else {
273                if (wordClass.matches(pattern1)) {
274                    temp = Integer.parseInt(wordClass.substring(0, 2));
275                    baseform = fullform.substring(0, fullform.length() - temp) + wordClass.substring(2);
276                } else if (wordClass.substring(i + 1).matches(pattern2)) {
277                    temp = Integer.parseInt(wordClass.substring(0, 1));
278                    if (temp == 0) {
279                        baseform = fullform;
280                    } else {
281                        temp = Integer.parseInt(wordClass.substring(0, 1));
282                        baseform = fullform.length() > temp
283                                ? fullform.substring(0, fullform.length() - temp) + wordClass.substring(1)
284                                : fullform;
285                    }
286                }
287            }
288        }
289        return baseform;
290    }

292    private String substitute(String word) {
293        return word.replace('‰', 'a')
294                .replace('ƒ', 'A')
295                .replace('¸', 'u')
296                .replace('‹', 'U')
297                .replace('ˆ', 'o')
298                .replace('÷', 'O');
299    }
