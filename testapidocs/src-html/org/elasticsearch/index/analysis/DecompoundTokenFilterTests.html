031public class DecompoundTokenFilterTests {
032
033    @Test
034    public void test() throws IOException {
035        AnalysisService analysisService = createAnalysisService();
036
037
038        TokenFilterFactory tokenFilter = analysisService.tokenFilter("decomp");
039        assertThat(tokenFilter, instanceOf(DecompoundTokenFilterFactory.class));
040
041        String source = "Die Jahresfeier der Rechtsanwaltskanzleien auf dem Donaudampfschiff hat viel Ökosteuer gekostet";
042
043        String[] expected = {
044            "Die",
045            "Die",
046            "Jahresfeier",
047            "Jahr",
048            "feier",
049            "der",
050            "der",
051            "Rechtsanwaltskanzleien",
052            "Recht",
053            "anwalt",
054            "kanzlei",
055            "auf",
056            "auf",
057            "dem",
058            "dem",
059            "Donaudampfschiff",
060            "Donau",
061            "dampf",
062            "schiff",
063            "hat",
064            "hat",
065            "viel",
066            "viel",
067            "Ökosteuer",
068            "Ökosteuer",
069            "gekostet",
070            "gekosten"
071        };
072
073        Tokenizer tokenizer = new StandardTokenizer(Version.LUCENE_36, new StringReader(source));
074
075        assertSimpleTSOutput(tokenFilter.create(tokenizer), expected);
076
077    }
078
079    public AnalysisService createAnalysisService() {
080        Settings settings = ImmutableSettings.settingsBuilder().loadFromClasspath("org/elasticsearch/index/analysis/decompound_analysis.json").build();
081
082        Index index = new Index("test");
083
084        Injector parentInjector = new ModulesBuilder().add(new SettingsModule(settings),
085                new EnvironmentModule(new Environment(settings)),
086                new IndicesAnalysisModule())
087                .createInjector();
088
089        AnalysisModule analysisModule = new AnalysisModule(settings, parentInjector.getInstance(IndicesAnalysisService.class));
090        new AnalysisDecompoundPlugin().onModule(analysisModule);
091
092        Injector injector = new ModulesBuilder().add(
093                new IndexSettingsModule(index, settings),
094                new IndexNameModule(index),
095                analysisModule)
096                .createChildInjector(parentInjector);
097
098        return injector.getInstance(AnalysisService.class);
099    }
100
101    public static void assertSimpleTSOutput(TokenStream stream,
102            String[] expected) throws IOException {
103        stream.reset();
104        CharTermAttribute termAttr = stream.getAttribute(CharTermAttribute.class);
105        Assert.assertNotNull(termAttr);
106        int i = 0;
107        while (stream.incrementToken()) {
108            Assert.assertTrue(i < expected.length);
109            Assert.assertEquals(expected[i++], termAttr.toString(), "expected different term at index " + i);
110        }
111        Assert.assertEquals(i, expected.length, "not all tokens produced");
112    }
113}
